3단계: 파일명 정리

Notion은 파일명에 ID를 추가합니다. 이를 제거하는 스크립트:

import os
import re

def clean_notion_filenames(directory):
    for root, dirs, files in os.walk(directory):
        for filename in files:
            if filename.endswith('.md'):
                # UUID 패턴 제거
                new_name = re.sub(r' [a-f0-9]{32}', '', filename)
                old_path = os.path.join(root, filename)
                new_path = os.path.join(root, new_name)
                os.rename(old_path, new_path)

clean_notion_filenames('./notion-export')

4단계: 링크 변환

Notion 링크를 옵시디언 형식으로 변환:

# Notion 링크: [페이지 이름](페이지%20이름%20abc123.md)
# Obsidian 링크: [[페이지 이름]]

import re

def convert_links(content):
    # URL 인코딩된 링크 변환
    pattern = r'\[([^\]]+)\]\(([^)]+)\.md\)'
    
    def replace_link(match):
        title = match.group(1)
        return f'[[{title}]]'
    
    return re.sub(pattern, replace_link, content)

# Evernote 태그를 옵시디언 태그로 변환
import xml.etree.ElementTree as ET

def convert_evernote_tags(enex_file):
    tree = ET.parse(enex_file)
    root = tree.getroot()
    
    for note in root.findall('.//note'):
        tags = []
        for tag in note.findall('.//tag'):
            tag_name = tag.text.replace(' ', '-')
            tags.append(f'#{tag_name}')
        
        # frontmatter에 태그 추가
        content = note.find('.//content').text
        frontmatter = f'---\ntags: [{", ".join(tags)}]\n---\n'
        
        return frontmatter + content

import json
from datetime import datetime

def convert_daily_notes(roam_json):
    with open(roam_json, 'r') as f:
        data = json.load(f)
    
    for page in data:
        title = page['title']
        
        # 날짜 형식인지 확인

            date = datetime.strptime(title, "%B %d, %Y")

깨진 링크 찾기
import os
import re

def find_broken_links(vault_path):
    broken_links = []
    all_files = set()
    
    # 모든 파일 목록 수집
    for root, dirs, files in os.walk(vault_path):
        for file in files:
            if file.endswith('.md'):
                all_files.add(file[:-3])  # .md 제거
    
    # 링크 검사
    for root, dirs, files in os.walk(vault_path):
        for file in files:
            if file.endswith('.md'):
                path = os.path.join(root, file)
                with open(path, 'r', encoding='utf-8') as f:
                    content = f.read()
                    
                    # [[링크]] 패턴 찾기
                    links = re.findall(r'\[\[([^\]]+)\]\]', content)
                    for link in links:
                        link_name = link.split('|')[0].split('#')[0]
                        if link_name not in all_files:
                            broken_links.append({
                                'file': file,
                                'broken_link': link
                            })
    
    return broken_links
중복 파일 제거
import hashlib
import os

def find_duplicates(vault_path):
    file_hashes = {}
    duplicates = []
    
    for root, dirs, files in os.walk(vault_path):
        for file in files:
            if file.endswith('.md'):
                path = os.path.join(root, file)
                
                # 파일 해시 계산
                with open(path, 'rb') as f:
                    file_hash = hashlib.md5(f.read()).hexdigest()
                
                if file_hash in file_hashes:
                    duplicates.append({
                        'original': file_hashes[file_hash],
                        'duplicate': path
                    })
                else:
                    file_hashes[file_hash] = path
    
    return duplicates
