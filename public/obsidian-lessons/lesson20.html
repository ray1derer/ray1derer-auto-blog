기본 플러그인 구조

import { Plugin } from 'obsidian';

export default class MyPlugin extends Plugin {
    async onload() {
        console.log('플러그인 로드됨');
        
        // 명령 추가
        this.addCommand({
            id: 'my-command',
            name: '내 명령 실행',
            callback: () => {
                console.log('명령 실행됨!');
            }
        });
        
        // 리본 아이콘 추가
        this.addRibbonIcon('dice', '내 플러그인', () => {
            console.log('아이콘 클릭됨!');
        });
    }
    
    onunload() {
        console.log('플러그인 언로드됨');
    }
}
