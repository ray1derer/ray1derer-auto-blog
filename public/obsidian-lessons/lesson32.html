const moods = {
    "😊": 0, "😐": 0, "😔": 0, "😡": 0, "😰": 0
}

dv.pages('"Journal"')
    .where(p => p.mood)
    .forEach(p => {
        if (moods[p.mood] !== undefined) {
            moods[p.mood]++
        }
    })

dv.header(3, "월간 감정 통계")
for (const [mood, count] of Object.entries(moods)) {
    const bar = "█".repeat(count)
    dv.paragraph(`${mood} ${bar} ${count}일`)
}
