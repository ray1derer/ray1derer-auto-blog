🤖 자동 작업 수집

<%*
// Templater: 일일 노트에서 작업 자동 수집
const today = tp.date.now("YYYY-MM-DD");
const tasks = app.vault.getMarkdownFiles()
  .filter(f => f.path.includes("Daily"))
  .map(async f => {
    const content = await app.vault.read(f);
    const taskRegex = /- \[ \] (.+)/g;
    const matches = [...content.matchAll(taskRegex)];
    return matches.map(m => ({
      task: m[1],
      source: f.basename
    }));
  });

const allTasks = (await Promise.all(tasks)).flat();
-%>
# 📋 Task Collector - <%= today %>

## Collected Tasks
<% allTasks.forEach(t => { %>
- [ ] <%= t.task %> (from: [[<%= t.source %>]])
<% }); %>

📊 진행률 추적

```dataviewjs
// 프로젝트별 진행률 차트
const projects = dv.pages('"1-Projects"')
  .where(p => p.status === "active");

const projectData = projects.map(p => {
  const tasks = p.file.tasks;
  const completed = tasks.where(t => t.completed).length;
  const total = tasks.length;
  const progress = total > 0 ? Math.round(completed / total * 100) : 0;
  
  return {
    name: p.file.name,
    progress: progress,
    completed: completed,
    total: total
  };
});

// 진행률 막대 그래프
dv.table(
  ["Project", "Progress", "Tasks"],
  projectData.map(p => [
    p.name,
    `${"█".repeat(Math.floor(p.progress/10))}${"░".repeat(10-Math.floor(p.progress/10))} ${p.progress}%`,
    `${p.completed}/${p.total}`
  ])
);
```

## 📈 Weekly Metrics
```dataviewjs
const lastWeek = [...Array(7)].map((_, i) => {
  const date = dv.date("today").minus({days: i});
  const tasks = dv.pages().file.tasks
    .where(t => t.completed && 
      dv.date(t.completion) && 
      dv.date(t.completion).toFormat("yyyy-MM-dd") === date.toFormat("yyyy-MM-dd"));
  
  return {
    date: date.toFormat("MM/dd"),
    count: tasks.length
  };
}).reverse();

dv.table(
  ["Date", "Completed", "Visual"],
  lastWeek.map(d => [
    d.date,
    d.count,
    "🟩".repeat(Math.min(d.count, 10))
  ])
);
```
