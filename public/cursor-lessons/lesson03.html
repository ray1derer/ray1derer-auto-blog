1. 코드 설명 요청하기

복잡한 코드를 이해하고 싶을 때:

// 복잡한 reduce 함수
const result = data
  .filter(item => item.active)
  .reduce((acc, item) => {
    const key = item.category;
    if (!acc[key]) {
      acc[key] = { items: [], total: 0 };
    }
    acc[key].items.push(item);
    acc[key].total += item.value;
    return acc;
  }, {});

이 코드가 뭘 하는지 설명해줘

이 코드는 데이터를 카테고리별로 그룹화하고 집계하는 작업을 수행합니다:

  1. 필터링: active가 true인 항목만 선택
  2. 그룹화: category를 키로 사용하여 객체로 변환
  3. 집계: 각 카테고리별로 items 배열과 total 합계 계산

예시 결과:

{
  "electronics": { items: [...], total: 1500 },
  "clothing": { items: [...], total: 800 }
}

2. 버그 수정 요청

에러가 발생한 코드를 수정하고 싶을 때:

// TypeError: Cannot read property 'map' of undefined
function renderList(items) {
  return items.map(item => <li>{item.name}</li>);
}

이 함수에서 "Cannot read property 'map' of undefined" 에러가 발생해. 어떻게 수정하면 될까?

items가 undefined일 때 발생하는 에러입니다. 다음과 같이 수정하세요:

function renderList(items) {
  // 방법 1: 기본값 설정
  return (items || []).map(item => <li>{item.name}</li>);
}

// 방법 2: 조건부 렌더링
function renderList(items) {
  if (!items || !Array.isArray(items)) {
    return <li>No items available</li>;
  }
  return items.map(item => <li>{item.name}</li>);
}

// 방법 3: 옵셔널 체이닝 (최신 문법)
function renderList(items) {
  return items?.map(item => <li>{item.name}</li>) || [];
}

3. 코드 리팩토링

기존 코드를 개선하고 싶을 때:

원본 코드

function calculatePrice(price, discount, tax) {
  let discountedPrice = price;
  if (discount > 0) {
    discountedPrice = price - (price * discount / 100);
  }
  let finalPrice = discountedPrice;
  if (tax > 0) {
    finalPrice = discountedPrice + (discountedPrice * tax / 100);
  }
  return Math.round(finalPrice * 100) / 100;
}

이 함수를 더 깔끔하고 읽기 쉽게 리팩토링해줘

함수를 더 간결하고 읽기 쉽게 리팩토링했습니다:

function calculatePrice(price, discount = 0, tax = 0) {
  const discountMultiplier = 1 - (discount / 100);
  const taxMultiplier = 1 + (tax / 100);
  
  const finalPrice = price * discountMultiplier * taxMultiplier;
  
  return Math.round(finalPrice * 100) / 100;
}

// 더 나아가서, 유틸리티 함수로 분리
const toPercentageMultiplier = (percentage, isDiscount = false) => 
  isDiscount ? 1 - (percentage / 100) : 1 + (percentage / 100);

const roundToTwoDecimals = (num) => 
  Math.round(num * 100) / 100;

function calculatePrice(price, discount = 0, tax = 0) {
  const afterDiscount = price * toPercentageMultiplier(discount, true);
  const afterTax = afterDiscount * toPercentageMultiplier(tax);
  
  return roundToTwoDecimals(afterTax);
}

4. 새 기능 구현

새로운 기능을 추가하고 싶을 때:

현재 TodoList 컴포넌트에 검색 기능을 추가하고 싶어. 실시간으로 필터링되도록 구현해줘.

TodoList에 실시간 검색 기능을 추가하겠습니다:

import React, { useState, useMemo } from 'react';

function TodoList({ todos }) {
  const [searchTerm, setSearchTerm] = useState('');
  
  // useMemo로 검색 결과 최적화
  const filteredTodos = useMemo(() => {
    if (!searchTerm.trim()) {
      return todos;
    }
    
    return todos.filter(todo =>
      todo.text.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }, [todos, searchTerm]);
  
  return (
    <div className="todo-list">
      <div className="search-box">
        <input
          type="text"
          placeholder="Search todos..."
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          className="search-input"
        />
        {searchTerm && (
          <button 
            onClick={() => setSearchTerm('')}
            className="clear-button"
          >
            Clear
          </button>
        )}
      </div>
      
      <div className="results-info">
        {searchTerm && (
          <p>Found {filteredTodos.length} of {todos.length} todos</p>
        )}
      </div>
      
      <ul className="todo-items">
        {filteredTodos.map(todo => (
          <TodoItem key={todo.id} todo={todo} />
        ))}
      </ul>
      
      {searchTerm && filteredTodos.length === 0 && (
        <p className="no-results">No todos found matching "{searchTerm}"</p>
      )}
    </div>
  );
}

효과적인 프롬프트 작성법

좋은 프롬프트 vs 나쁜 프롬프트

❌ 피해야 할 프롬프트

  • "코드 고쳐줘" - 너무 모호함
  • "에러 났어" - 정보 부족
  • "더 좋게 만들어줘" - 구체성 없음

✅ 권장하는 프롬프트

  • "이 함수의 시간 복잡도를 O(n)으로 개선해줘"
  • "useState 대신 useReducer를 사용하도록 리팩토링해줘"
  • "이 API 호출에 에러 처리와 로딩 상태를 추가해줘"
