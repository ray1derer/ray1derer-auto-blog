옵션 1: 두 기능 모두 통합
function calculateTotal(items, options = {}) {
    const { includeTax = true, applyDiscount = true } = options;
    
    const subtotal = items.reduce((sum, item) => 
        sum + (item.price * item.quantity), 0);
    
    let total = subtotal;
    
    if (includeTax) {
        const tax = subtotal * 0.1;
        total += tax;
    }
    
    if (applyDiscount && subtotal > 100) {
        const discount = subtotal * 0.05;
        total -= discount;
    }
    
    return total;
}

두 브랜치의 기능을 모두 포함하며, 옵션으로 제어 가능

옵션 2: 전략 패턴 적용
class PricingStrategy {
    calculateTax(subtotal) { return 0; }
    calculateDiscount(subtotal) { return 0; }
}

class StandardPricing extends PricingStrategy {
    calculateTax(subtotal) { return subtotal * 0.1; }
}

class DiscountPricing extends PricingStrategy {
    calculateDiscount(subtotal) {
        return subtotal > 100 ? subtotal * 0.05 : 0;
    }
}

function calculateTotal(items, strategy = new StandardPricing()) {
    const subtotal = items.reduce((sum, item) => 
        sum + (item.price * item.quantity), 0);
    
    const tax = strategy.calculateTax(subtotal);
    const discount = strategy.calculateDiscount(subtotal);
    
    return subtotal + tax - discount;
}

Git Aliases for Cursor

# ~/.gitconfig
[alias]
    # AI 커밋
    aic = !cursor commit
    
    # AI 리뷰와 함께 푸시
    pushrev = !git push && cursor review
    
    # 스마트 머지
    smerge = !cursor merge
    
    # 충돌 해결
    resolve = !cursor resolve-conflicts
