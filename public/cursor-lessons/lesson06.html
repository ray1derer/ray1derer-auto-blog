import { useState, useEffect, useCallback, useRef } from 'react';

interface UseInfiniteScrollOptions {
  threshold?: number;
  rootMargin?: string;
  onLoadMore: () => Promise;
  hasMore: boolean;
}

export const useInfiniteScroll = ({
  threshold = 0.1,
  rootMargin = '0px',
  onLoadMore,
  hasMore
}: UseInfiniteScrollOptions) => {
  const [isLoading, setIsLoading] = useState(false);
  const observerRef = useRef(null);
  
  const lastElementRef = useCallback((node: HTMLElement | null) => {
    if (isLoading) return;
    
    if (observerRef.current) {
      observerRef.current.disconnect();
    }
    
    if (!hasMore || !node) return;
    
    observerRef.current = new IntersectionObserver(
      async (entries) => {
        if (entries[0].isIntersecting && hasMore && !isLoading) {
          setIsLoading(true);
          try {
            await onLoadMore();
          } catch (error) {
            console.error('Error loading more items:', error);
          } finally {
            setIsLoading(false);
          }
        }
      },
      { threshold, rootMargin }
    );
    
    observerRef.current.observe(node);
  }, [isLoading, hasMore, onLoadMore, threshold, rootMargin]);
  
  useEffect(() => {
    return () => {
      if (observerRef.current) {
        observerRef.current.disconnect();
      }
    };
  }, []);
  
  return { lastElementRef, isLoading };
};

Claude의 응답
import { useState, useEffect, useCallback, useRef } from 'react';

/**
 * 무한 스크롤을 위한 커스텀 훅
 * 
 * @param {Function} loadMore - 추가 데이터를 로드하는 함수
 * @param {boolean} hasMore - 더 로드할 데이터가 있는지 여부
 * @param {Object} options - 추가 옵션
 * @returns {Object} - 관찰할 요소의 ref와 로딩 상태
 */
function useInfiniteScroll(loadMore, hasMore, options = {}) {
  const {
    threshold = 0.5,      // 요소가 얼마나 보여야 트리거되는지 (0-1)
    rootMargin = '100px', // 뷰포트 확장 영역
    enabled = true        // 훅 활성화 여부
  } = options;

  const [loading, setLoading] = useState(false);
  const observerRef = useRef();
  const loadingRef = useRef(loading);
  
  // loading 상태를 ref로도 추적 (클로저 문제 해결)
  useEffect(() => {
    loadingRef.current = loading;
  }, [loading]);

  // 관찰 대상 요소에 연결할 콜백 ref
  const targetRef = useCallback((node) => {
    // 기존 observer 정리
    if (observerRef.current) {
      observerRef.current.disconnect();
    }

    // 비활성화 상태거나 더 이상 로드할 것이 없으면 중단
    if (!enabled || !hasMore || !node) return;

    // IntersectionObserver 생성
    observerRef.current = new IntersectionObserver(
      async (entries) => {
        const [entry] = entries;
        
        // 요소가 보이고 로딩 중이 아닐 때만 실행
        if (entry.isIntersecting && !loadingRef.current) {
          setLoading(true);
          
          try {
            await loadMore();
          } catch (error) {
            console.error('무한 스크롤 로딩 중 오류:', error);
            // 에러 발생 시에도 다시 시도할 수 있도록 loading을 false로
          } finally {
            setLoading(false);
          }
        }
      },
      {
        threshold,
        rootMargin
      }
    );

    // 새 요소 관찰 시작
    observerRef.current.observe(node);
  }, [loadMore, hasMore, enabled, threshold, rootMargin]);

  // 컴포넌트 언마운트 시 정리
  useEffect(() => {
    return () => {
      if (observerRef.current) {
        observerRef.current.disconnect();
      }
    };
  }, []);

  return {
    targetRef,  // 마지막 요소에 연결할 ref
    loading,    // 로딩 상태
    setLoading  // 수동으로 로딩 상태를 제어할 수 있도록
  };
}

모델별 특성 이해

각 AI 모델은 고유한 강점이 있으며, 작업의 성격에 따라 적절히 선택해야 합니다.

GPT-4는 복잡한 작업에

높은 추론 능력이 필요한 복잡한 로직이나 창의적인 솔루션이 필요할 때 선택합니다.

Claude는 분석과 설명에
